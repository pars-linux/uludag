 mkdir cvsexample2

 rm -rf cvsexample/* 
 cd ~/cvsexample
 cvs import -m "dir structure" cvsexample yourname start 
 cd ..
 rm -r cvsexample 
 cvs checkout cvsexample 
 ls
\end{verbatim}

   svn diff
 \end{verbatim}
